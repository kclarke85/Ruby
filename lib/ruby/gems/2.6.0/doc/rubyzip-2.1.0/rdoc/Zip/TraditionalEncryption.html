def reset_keys!
  @key0 = 0x12345678
  @key1 = 0x23456789
  @key2 = 0x34567890
  @password.each_byte do |byte|
    update_keys(byte.chr)
  end
end

def update_keys(n)
  @key0 = ~Zlib.crc32(n, ~@key0)
  @key1 = ((@key1 + (@key0 & 0xff)) * 134_775_813 + 1) & 0xffffffff
  @key2 = ~Zlib.crc32((@key1 >> 24).chr, ~@key2)
end
