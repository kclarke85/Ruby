def initial_parse(binstr)
  if !binstr
    # If nil, start with empty.
    return false
  elsif binstr[0, 2] != self.class.const_get(:HEADER_ID)
    warn 'WARNING: weird extra field header ID. Skip parsing it.'
    return false
  end
  [binstr[2, 2].unpack('v')[0], binstr[4..-1]]
end
