def initialize(zipfile, entry, srcPath = nil, permissionInt = nil)
  super(zipfile, entry)

  @ftype = :directory
  entry.get_extra_attributes_from_path(srcPath) if srcPath
  @unix_perms = permissionInt if permissionInt
end
