def merge(binstr)
  return if binstr.empty?
  size, content = initial_parse(binstr)
  # size: 0 for central directory. 4 for local header
  return if !size || size == 0
  atime, mtime, uid, gid = content.unpack('VVvv')
  @uid ||= uid
  @gid ||= gid
  @atime ||= atime
  @mtime ||= mtime
end
