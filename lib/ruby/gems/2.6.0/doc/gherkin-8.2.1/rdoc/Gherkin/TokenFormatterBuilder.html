def format_token(token)
  return "EOF" if token.eof?

  sprintf "(%s:%s)%s:%s/%s/%s",
    token.location[:line],
    token.location[:column],
    token.matched_type,
    token.matched_keyword,
    token.matched_text,
    Array(token.matched_items).map { |i| "#{i.column}:#{i.text}"}.join(',')
end
