def self.encode_source_message(uri, data)
  Cucumber::Messages::Source.new({
    uri: uri,
    data: data,
    media: Cucumber::Messages::Media.new({
      encoding: :UTF8,
      content_type: 'text/x.cucumber.gherkin+plain'
    })
  })
end
