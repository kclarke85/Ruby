def initialize(args)
  @host = args['host']
  @port = args['port']
  @unix = args['unix'] if RUBY_PLATFORM !~ /mingw|mswin/
  @timeouts = DEFAULT_TIMEOUTS.merge(args['timeout'] || {})
end
